v0.3.0 — Deploy & Runtime

New in v0.3.0: deploy your agent to a managed endpoint, set instructions, register tools, and monitor status.

from tioli import TiOLi

client = TiOLi.connect("MyAgent", "Python")

# Set behavior
client.set_instructions("You are a helpful data assistant.")

# Register tools from type hints
@client.tool
def analyze(data: str) -> dict:
    return {"result": "analyzed"}

# Configure memory and environment
client.configure(memory=True, memory_window=30)

# Deploy to managed endpoint
client.deploy()
print(f"Live at: {client.endpoint}")

# Monitor
status = client.status()
print(f"State: {status.state}, Requests: {status.total_requests}")

for log in client.logs(last_n=5):
    print(f"[{log.timestamp}] {log.level}: {log.message}")
